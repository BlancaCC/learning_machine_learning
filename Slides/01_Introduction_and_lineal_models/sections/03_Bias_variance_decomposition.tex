\begin{frame}{Bias Variance trade off}
\section{The Bias-Variance Decomposition}
\subsection{The bias variance trade-off}

\begin{align}
    (y(x)-t)^2 
    &=  
    (y(x) -\E[t|x] + \E[t|x] -t)^2  
    \\
    &= 
    (y(x) -\E[t|x])^2 
    + 
    (\E[t|x] -t)^2 
    +
    2(y(x) -\E[t|x])(\E[t|x] -t)
\end{align}

Substituting into the loss function 

\begin{align}
    \E[L] &= 
    \int \int
    p(x,t) (y(x)-t)^2 
    dt dx
\end{align}

Performing the integral over $t$, we see that the cross-term
vanishes and we obtain 

\begin{align}
    \E[L]
    &=  
    \int 
    (y(x) -\E[t|x])^2 p(x) dx
    + 
    \int 
    (\E[t|x] -t)^2 p(x) dx.
\end{align}
\end{frame}

\begin{frame}
    \begin{align}
        \E[L]
        &=  
        \int 
        (y(x)- E[y] + E[y] -\E[t|x])^2 p(x) dx
        + 
        \int 
        (\E[t|x] -t)^2 p(x) dx
        \nonumber
        \\
        &=  
        \int 
        (E[y] -\E[t|x])^2 p(x) dx
        +
        \int 
        (y(x)- E[y])^2 p(x) dx
        + 
        \int 
        (\E[t|x] -t)^2 p(x) dx
        \nonumber
        \\
        &=
        \text{Bias}^2 + Var(y(x)) + \text{noise}.
    \end{align}
\end{frame}
